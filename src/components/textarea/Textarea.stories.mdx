import { Canvas, Meta, Story, ArgsTable, Source } from "@storybook/blocks";
import Textarea from "./Textarea";
import { TEXTAREA_KIND, TEXTAREA_SIZE } from "./types";
import { useState } from "react";
import { useStyletron } from "baseui";

<Meta title="Form/Textarea" component={Textarea} />

export const Template = ({ value: baseValue, ...args }) => {
  const [value, setValue] = useState(baseValue);
  const [css] = useStyletron();
  const onChangeHandler = (event) => {
    setValue(event.target.value);
  };
  return (
    <div className={css({ maxWidth: "400px" })}>
      <Textarea {...args} value={value} onChange={onChangeHandler} />
    </div>
  );
};

### Usage:

To use, import the component `Textarea` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Textarea, TEXTAREA_SIZE} from "@nilfoundation/ui-kit";
        ...
        <Textarea placeholder="Textarea Field" size={TEXTAREA_SIZE.small} />
        // Disabled
        <Textarea placeholder="Textarea Field" disabled />
        // Readonly
        <Textarea placeholder="Textarea Field" readOnly />
        // Loading
        <Textarea placeholder="Textarea Field" isLoading />
        // Clearable
        <Textarea placeholder="Textarea Field" clearable />
        // Resizable
        <Textarea placeholder="Textarea Field" resize="both" />
  `}
/>
