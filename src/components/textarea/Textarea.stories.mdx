### Usage:

To use, import the component `Textarea` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Textarea, TEXTAREA_SIZE} from "@nilfoundation/ui-kit";
        ...
        <Textarea placeholder="Textarea Field" size={TEXTAREA_SIZE.small} />
        // Disabled
        <Textarea placeholder="Textarea Field" disabled />
        // Readonly
        <Textarea placeholder="Textarea Field" readOnly />
        // Loading
        <Textarea placeholder="Textarea Field" isLoading />
        // Clearable
        <Textarea placeholder="Textarea Field" clearable />
        // Resizable
        <Textarea placeholder="Textarea Field" resize="both" />
  `}
/>
