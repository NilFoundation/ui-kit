### Usage:

To use, import the component `Badge` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Badge, BADGE_COLOR, BADGE_HIERARCHY, BADGE_PLACEMENT} from "@nilfoundation/ui-kit";
        ...
         <Badge content="Badge" />
        // With color
         <Badge content="Badge" color={BADGE_COLOR.positive} />
        // With hierarchy
        <Badge content="Badge" hierarchy={BADGE_HIERARCHY.low} />
        // With children
        <Badge content="Badge">
            <div>...</div>
        </Badge>
        // With placement
        <Badge content="Badge" placement={BADGE_PLACEMENT.bottomLeft}>
            <div>...</div>
        </Badge>
  `}
/>
