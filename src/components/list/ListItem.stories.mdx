### Usage:

To use, import the component `ListItem` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {ListItem, ListItemLabel, LIST_ARTWORK_SIZES, Button, HeartIcon} from "@nilfoundation/ui-kit";
        ...
        <ListItem>
          <ListItemLabel description="Description">Label</ListItemLabel>
        </ListItem>
        // Active
        <ListItem isActive>
          <ListItemLabel description="Description">Label</ListItemLabel>
        </ListItem>
        // Disabled
        <ListItem disabled>
          <ListItemLabel description="Description">Label</ListItemLabel>
        </ListItem>
         // With end enhancer
        <ListItem endEnhancer={(props) => <Button {...props}>Action</Button>}>
          <ListItemLabel description="Description">Label</ListItemLabel>
        </ListItem>
         // With artwork
        <ListItem artwork={(props) => <HeartIcon {...props} />} artworkSize={LIST_ARTWORK_SIZES.SMALL}>
          <ListItemLabel description="Description">Label</ListItemLabel>
        </ListItem>
  `}
/>
