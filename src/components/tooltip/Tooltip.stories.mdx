### Usage:

To use, import the component `Tooltip` or `StatefulTooltip` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {StatefulTooltip, Tooltip} from "@nilfoundation/ui-kit";
        ...
        <StatefulTooltip content="StatefulTooltip content" showArrow>
          <span>
            Hover me
          </span>
        </StatefulTooltip>
        // Tooltip on the left side of the button
        <StatefulTooltip content="StatefulTooltip content" showArrow placement="left">
            ...
        </StatefulTooltip>
        // Tooltip's triggering event is click
        <StatefulTooltip content="StatefulTooltip content" triggerType="click" showArrow>
            ...
        </StatefulTooltip>
        // Tooltip usage
        <Tooltip content="Tooltip content" showArrow isOpen>
            ...
        </Tooltip>
      `}
/>
