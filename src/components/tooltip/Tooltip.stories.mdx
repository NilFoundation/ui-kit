### Usage:

To use, import the component `Tooltip` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Tooltip} from "@nilfoundation/ui-kit";
        ...
        <Tooltip content="Tooltip content" showArrow>
          <span>
            Hover me
          </span>
        </Tooltip>
        // Tooltip on the left side of the button
        <Tooltip content="Tooltip content" showArrow placement="left">
            ...
        </Tooltip>
        // Tooltip's triggering event is click
        <Tooltip content="Tooltip content" triggerType="click" showArrow>
            ...
        </Tooltip>
      `}
/>
