import { Canvas, Meta, Story, ArgsTable, Source } from "@storybook/blocks";
import { useState } from "react";
import RadioGroup, { ALIGN } from "./RadioGroup";
import Radio from "./Radio";

<Meta title="Form/Radio" component={RadioGroup} />

export const Template = ({ value: baseValue, ...args }) => {
  const [value, setValue] = useState(baseValue);
  const onChangeHandler = (event) => {
    setValue(event.target.value);
  };
  return (
    <RadioGroup {...args} value={value} onChange={onChangeHandler}>
      <Radio value="1">One</Radio>
      <Radio value="2" description="This is a radio description">
        Two
      </Radio>
      <Radio value="3">Three</Radio>
    </RadioGroup>
  );
};

### Usage:

To use, import the component `Radio` and `RadioGroup` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {RadioGroup, Radio} from "@nilfoundation/ui-kit";
        ...
        <RadioGroup align={ALIGN.horizontal} name="name">
          <Radio value="1">One</Radio>
          <Radio value="2">Two</Radio>
          <Radio value="3">Three</Radio>
        </RadioGroup>
        <RadioGroup value="2" align={ALIGN.horizontal} name="name">
          <Radio value="1">One</Radio>
          <Radio value="2">Two</Radio>
          <Radio value="3">Three</Radio>
        </RadioGroup>
        <RadioGroup error align={ALIGN.horizontal} name="name">
          <Radio value="1">One</Radio>
          <Radio value="2">Two</Radio>
          <Radio value="3">Three</Radio>
        </RadioGroup>
        <RadioGroup disabled align={ALIGN.horizontal} name="name">
          <Radio value="1">One</Radio>
          <Radio value="2">Two</Radio>
          <Radio value="3">Three</Radio>
        </RadioGroup>
    `}
/>
