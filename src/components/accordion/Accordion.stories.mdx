### Usage:

To use, import the component `Accordion` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Accordion, Panel, Filter} from "@nilfoundation/ui-kit";
        ...
        <Accordion>
            <Panel title="Panel 1">Content 1</Panel>
            <Panel title="Panel 2">Content 2</Panel>
            <Panel title="Panel 3">Content 3</Panel>
        </Accordion>
        // With icon
        <Accordion>
            <Panel title="Panel 1" icon={<Filter />}>Content 1</Panel>
            <Panel title="Panel 2" icon={<Filter />}>Content 2</Panel>
            <Panel title="Panel 3" icon={<Filter />}>Content 3</Panel>
        </Accordion>
        // With description
        <Accordion>
            <Panel title="Panel 1" description="Comment 1">Content 1</Panel>
            <Panel title="Panel 2" description="Comment 2">Content 2</Panel>
            <Panel title="Panel 3" description="Comment 3">Content 3</Panel>
        </Accordion>
  `}
/>
