### Usage:

To use, import the component `ProgressBar` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {ProgressBar, PROGRESS_BAR_SIZE} from "@nilfoundation/ui-kit";
        ...
        <ProgressBar value={48} />
        // With size
        <ProgressBar value={48} size={PROGRESS_BAR_SIZE.large} />
        // With minValue, maxValue
        <ProgressBar value={400} minValue={100} maxValue={1000} />
        // With label
         <ProgressBar value={48} getProgressLabel={(value, minValue, maxValue) => ''} showLabel />
        // With error message
        <ProgressBar value={48} errorMessage="Error message" showLabel />
  `}
/>
