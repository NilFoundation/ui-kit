### Usage:

To use, import the component `Menu` from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Menu, MENU_SIZE, SearchIcon} from "@nilfoundation/ui-kit";
        const ITEMS = [
          { label: "Item One" },
          { label: "Item Two" },
          { label: "Item Three" },
          { label: "Item Four" }
        ]
        ...
        <Menu items={[ITEMS]} />
        // To use a checkbox, you must define a selected property with values true or false other than null
        <Menu size={MENU_SIZE.small} items={[...ITEMS, { label: "Item Fifth", selected: false}]} />
        // Disabled item
        <Menu size={MENU_SIZE.small} items={[...ITEMS, { label: "Item Fifth", disabled: true}]} />
        // With start enhancer
        <Menu size={MENU_SIZE.small} items={[...ITEMS, { label: "Item Fifth", startEnhancer: <SearchIcon />}]} />
         // With end enhancer
        <Menu size={MENU_SIZE.small} items={[...ITEMS, { label: "Item Fifth", endEnhancer: <SearchIcon />}]} />
        // With suffix text
        <Menu size={MENU_SIZE.small} items={[...ITEMS, { label: "Item Fifth", suffixText: "Text"}]} />
  `}
/>
