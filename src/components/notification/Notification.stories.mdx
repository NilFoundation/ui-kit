### Usage:

To use, import the component `Notification` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Notification, NOTIFICATION_KIND, Search} from "@nilfoundation/ui-kit";
        ...
        <Notification>This is a notification.</Notification>
        // With kind
        <Notification kind={NOTIFICATION_KIND.positive}>This is a notification.</Notification>
        // Without icon
        <Notification hideIcon>This is a notification.</Notification>
        // Without custom icon
        <Notification icon={<Search size={20} />}>This is a notification.</Notification>
        // Closable
        <Notification closeable>This is a notification.</Notification>
        // With dismiss action
        <Notification isAction actionLabel="Close">This is a notification.</Notification>
        // Auto hide after 3000ms
        <Notification autoHideDuration={3000}>This is a notification.</Notification>
  `}
/>
