### Usage:

To use, import the component `Drawer` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        ...
        import {Drawer, DRAWER_ANCHOR, DRAWER_SIZE} from "@nilfoundation/ui-kit";
        ...
        const [isOpen, setOpen] = useState(false);
        ...
        <Drawer isOpen={isOpen} onClose={() => setOpen(false)}>
            <p>Drawer content</p>
        </Drawer>
        // Anchor
        <Drawer anchor={DRAWER_ANCHOR.top} isOpen={isOpen} onClose={() => setOpen(false)}>
            <p>Drawer content</p>
        </Drawer>
        // Size
        <Drawer anchor={DRAWER_SIZE.full} isOpen={isOpen} onClose={() => setOpen(false)}>
            <p>Drawer content</p>
        </Drawer>
    `}
/>
