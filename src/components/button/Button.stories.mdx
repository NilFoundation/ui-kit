### Usage:

To use, import the component `Button` and from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Button, BUTTON_SIZE, BUTTON_SHAPE, BUTTON_KIND Upload} from "@nilfoundation/ui-kit";
         ...
        <Button>Button Label</Button>
        // With size
        <Button size={BUTTON_SIZE.mini}>Button Label</Button>
        // With kind
        <Button kind={BUTTON_KIND.gray}>Button Label</Button>
        // With shape
        <Button shape={BUTTON_SHAPE.square}>Button Label</Button>
        // Disabled
        <Button disabled>Button Label</Button>
        // Loading
        <Button isLoading>Button Label</Button>
        // With enhancer
        <Button startEnhancer={<Upload />} endEnhancer={<Upload />}>Button Label</Button>
  `}
/>
