### Usage:

To use, import the component `NavigationBar` and from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {NavigationBar, Brand, AuthDropdownContainer, AuthBalance} from "@nilfoundation/ui-kit";
        ...
        export const items = [
          {
            id: "1",
            label: "Menu item 1",
            isSelected: true
          },
          {
          id: "2",
          label: "Menu item 2",
          children: [
              {
              id: "3",
              label: "Menu subitem 1"
              }
            ]
          }
        ]
        ...
        <NavigationBar
            items={items}
            username="John"
            isAuth
            brand={<Brand />}
            authDropdownContainer={() => (
              <AuthDropdownContainer username="John">
                <AuthBalance amount="1069" coin="34" currency="USD" />
              </AuthDropdownContainer>
            )}
        />
  `}
/>
