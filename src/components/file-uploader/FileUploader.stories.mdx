### Usage:

To use, import the component `FileUploader` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {FileUploader} from "@nilfoundation/ui-kit";
        ...
        <FileUploader />
        // Disabled
        <FileUploader disabled />
        // Multiple
        <FileUploader multiple />
        // With error message
        <FileUploader errorMessage="Upload failed... connection was lost." />
        // With progress message
        <FileUploader progressAmount={64} progressMessage="Uploading... 64% of 100%" />
  `}
/>
