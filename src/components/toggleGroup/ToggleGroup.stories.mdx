import { Canvas, Meta, Story, ArgsTable, Source } from "@storybook/addon-docs/blocks";
import { useState } from "react";
import ToggleGroup from "./ToggleGroup";
import { getFigmaParameters } from "../../shared/ui/getFigmaParameters";

<Meta title="Form/ToggleGroup" component={ToggleGroup} />

export const Template = ({ ...args }) => {
  const [selectedOptions, setSelectedOptions] = useState([]);
  return (
    <div>
      <ToggleGroup
        {...args}
        value={selectedOptions}
        onChange={(options) => {
          setSelectedOptions(options);
        }}
      />
    </div>
  );
};

### Usage:

To use, import the component `ToggleGroup` and from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {Button, BUTTON_SIZE, BUTTON_SHAPE, BUTTON_KIND Upload} from "@nilfoundation/ui-kit";
         ...
        <ToggleGroup
          options={[{ label: "Proof Market", key: "proof-market" }]}
          selectedOptions={selectedOptions}
          onToggle={(options) => {
            setSelectedOptions(options);
          }}
        />
  `}
/>
