### Usage:

To use, import the components `Tabs` and `Tab` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Tabs, Tab, TABS_ORIENTATION, SearchIcon} from "@nilfoundation/ui-kit";
        ...
        <Tabs>
          <Tab title="Tab Link 1" startEnhancer={<SearchIcon />} endEnhancer={<SearchIcon />}>
            <span>Tab Content 1</span>
          </Tab>
          <Tab title="Tab Link 2">
            <span>Tab Content 2</span>
          </Tab>
          <Tab title="Tab Link 3">
            <span>Tab Content 3</span>
          </Tab>
        </Tabs>
        // Disabled
        <Tabs disabled>
          ...
        </Tabs>
        // With initial active key
        <Tabs activeKey="2">
          ...
        </Tabs>
        // Vertical
        <Tabs orientation={TABS_ORIENTATION.vertical}>
          ...
        </Tabs>
  `}
/>
