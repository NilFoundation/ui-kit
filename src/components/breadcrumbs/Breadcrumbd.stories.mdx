### Usage:

To use, import the component `Breadcrumbs` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Breadcrumbs, BreadcrumbsItem} from "@nilfoundation/ui-kit";
        ...
        <Breadcrumbs>
          <BreadcrumbsItem href="#">Parent Page</BreadcrumbsItem>
          <BreadcrumbsItem href="#">Sub-Parent Page</BreadcrumbsItem>
          <BreadcrumbsItem isActive>Current Page</BreadcrumbsItem>
        </Breadcrumbs>
        // With disabled item
        <Breadcrumbs>
          <BreadcrumbsItem disabled href="#">Parent Page</BreadcrumbsItem>
          <BreadcrumbsItem href="#">Sub-Parent Page</BreadcrumbsItem>
          <BreadcrumbsItem isActive>Current Page</BreadcrumbsItem>
        </Breadcrumbs>
  `}
/>
