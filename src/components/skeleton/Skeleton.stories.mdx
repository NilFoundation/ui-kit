### Usage:

To use, import the component `Skeleton` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Skeleton, SKELETON_SIZE} from "@nilfoundation/ui-kit";
        const WIDTH = "400px";
        const HEIGHT = "400px";
        ...
        <Skeleton width={WIDTH} height={HEIGHT} />
        // With animation
        <Skeleton width={WIDTH} height={HEIGHT} animation />
        // With size
         <Skeleton width={WIDTH} height={HEIGHT} animation size={SKELETON_SIZE.medium} />
        // With rows
        <Skeleton width={WIDTH} height={HEIGHT} animation rows={4} />
  `}
/>
