### Usage:

To use, import the component `ButtonIcon` and from `@nilfoundation/ui-kit`.

<Source
  language="tsx"
  dark
  format={true}
  code={`
        import {ButtonIcon, BUTTON_SIZE, BUTTON_KIND} from "@nilfoundation/ui-kit";
         ...
        <ButtonIcon icon={<HeartIcon />} />
        // With size
        <ButtonIcon size={BUTTON_SIZE.large} icon={<HeartIcon />} />
        // With kind
        <Button kind={BUTTON_KIND.gray} icon={<HeartIcon />}>Button Label</Button>
        // Disabled
        <Button disabled icon={<HeartIcon />}>Button Label</Button>
  `}
/>
