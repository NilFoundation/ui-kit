### Usage:

To use, import the component `Card` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Card} from "@nilfoundation/ui-kit";
        ...
        // Base card with the description
        <Card> Description </Card>
        // Card with the description, border and title
        <Card border title="Card title"> Description </Card>
        // Card with the button
        <Card button="Button text"> Description </Card>
        // Card with the dash on the card's top
        <Card dash> Description </Card>
        // Card with header image
        <Card headerImage="path to image"> Description </Card>
        // Card with thumbnail
        <Card thumbnail="path to image"> Description </Card>
  `}
/>
