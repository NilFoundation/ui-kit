### Usage:

To use, import the component `Card` from `@nilfoundation/ui-kit`.

<Source
  language='tsx'
  dark
  format={true}
  code={`
        import {Card, StyledBody, StyledAction, Button} from "@nilfoundation/ui-kit";
        ...
        <Card>
          <StyledBody>
            Proin ut dui sed metus pharetra hend rerit vel non mi. Nulla ornare faucibus ex, non facilisis nisl.
          </StyledBody>
        </Card>
        // Card border
        <Card border title="Card title">...</Card>
        // With action
        <Card>
          <StyledAction>
            <Button>Button label</Button>
          </StyledAction>
        </Card>
        // With headline
        <Card headline>Description</Card>
        // With header image
        <Card headerImage="path">...</Card>
        // With thumbnail
        <Card thumbnail="path">...</Card>
  `}
/>
